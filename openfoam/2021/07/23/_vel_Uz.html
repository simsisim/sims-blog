def ustar(uref, zref, z0):
    """Calculate friction velocity for a neutral ABL logartithmic profile."""
    return KARMAN * uref / log((zref + z0) / z0)


def u_z(z, z0, zref, uref):
    frac = ustar(uref, zref, z0)/ KARMAN
    _ = frac * np.log(z/ z0)
    return _.reshape(-1, 1)
